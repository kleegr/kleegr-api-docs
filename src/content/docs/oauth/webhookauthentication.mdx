## Verification example (Node.js)

```javascript
const crypto = require('crypto');

const PUBLIC_KEY = `-----BEGIN PUBLIC KEY-----
... (key above) ...
-----END PUBLIC KEY-----`;

function verifyWebhook(rawBody, signature) {
  const verifier = crypto.createVerify('SHA256');
  verifier.update(rawBody);
  return verifier.verify(PUBLIC_KEY, signature, 'base64');
}

app.post('/webhook', (req, res) => {
  const sig = req.headers['x-wh-signature'];
  if (!verifyWebhook(JSON.stringify(req.body), sig)) {
    return res.status(401).send('Invalid signature');
  }
  // Process the webhook...
  res.sendStatus(200);
});
```

## Prevent replay attacks

- Reject requests where `timestamp` is older than 5 minutes
- Track and reject duplicate `webhookId` values
